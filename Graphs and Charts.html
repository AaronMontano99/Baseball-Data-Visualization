import pandas as pd
import matplotlib.pyplot as plt

# Load the dataset and set "AVG" as the index column
hitting_stats = pd.read_csv("baseball_hitting.csv", index_col="AVG")

# Sort the dataset by "AVG" in descending order
sorted_stats = hitting_stats.sort_values(by="AVG", ascending=False)

# Select the top 15 rows
top_15 = sorted_stats.head(15)

# Print the names of the players
print(top_15["Player name"])

# Create a bar chart with player names and their At-bat vs Hits
plt.figure(figsize=(10, 6))
plt.bar(range(len(top_15)), top_15["At-bat"], label="At-bat")
plt.bar(range(len(top_15)), top_15["Hits"], label="Hits")

# Set the x-axis ticks and labels
plt.xticks(range(len(top_15)), top_15.index, rotation='vertical')

# Set the chart title and axis labels
plt.title("Top 15 Players - At-bat vs Hits")
plt.xlabel("Batting Avg")
plt.ylabel("At-Bats")

# Add legend and adjust layout
plt.legend()
plt.tight_layout()

# Display the chart
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Load the dataset
pitcher_stats = pd.read_csv("baseball_pitcher.csv")

# Filter the dataset to include only pitchers with at least 600 innings pitched
filtered_stats = pitcher_stats[pitcher_stats["Innings pitched"] >= 600]

# Select the top 15 rows based on "Complete games"
top_15 = filtered_stats.nlargest(15, "Complete Game")

# Create a grouped bar chart with player names and their Complete Games vs Games Started
plt.figure(figsize=(10, 6))

# Set the bar width
bar_width = 0.35

# Position of the bars on the x-axis
r1 = range(len(top_15))
r2 = [x + bar_width for x in r1]

# Plot Complete Games and Games Started using grouped bars
plt.bar(r1, top_15["Complete Game"], color='blue', width=bar_width, label='Complete Games')
plt.bar(r2, top_15["Games Started"], color='red', width=bar_width, alpha=0.5, label='Games Started')

# Set the x-axis ticks and labels
plt.xticks(range(len(top_15)), top_15["Player name"], rotation='vertical')

# Set the chart title and axis labels
plt.title("Top 15 Pitchers - Complete Games vs Games Started")
plt.xlabel("Player")
plt.ylabel("Count")

# Add legend
plt.legend()

# Add pitcher information separately
print("Pitcher Information:")
for _, row in top_15.iterrows():
    print("- Player: " + row["Player name"])
    print("  - Complete Games: " + str(row["Complete Game"]))
    print("  - Games Started: " + str(row["Games Started"]))
    print()

# Display the chart
plt.tight_layout()
plt.show()
